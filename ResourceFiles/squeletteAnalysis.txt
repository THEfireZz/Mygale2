@echo off
setlocal enabledelayedexpansion

set "folder=<dossierImages>\"
set "start=<firstImg>"
set "end=<lastImg>"
set "extension=.<format>"
set "image=<name>"
set count=0

for /l %%i in (%start%,1,%end%) do (
   set num=000%%i
   set num=!num:~-4!
   set "filename=%folder%%image%_!num!%extension%"
   echo !filename!
   if not exist "!filename!" (
        set listeImages=!listeImages!%%i,
        set /a count+=1
 rem       set %%i=+1
   )
)
echo count : %count%
set resubmissionCount=%count%
echo !listeImages! > <path>mygale\TEMP\<jobName>\lsf\missingIndice.txt
